   6:  * Binary Decision Diagram interface
   7:  * 
   8:  * Binary decision diagrams are data structures that are used to represent
   9:  * boolean functions. This project aims to implement a module that can be
  10:  * utilized in creating BDDs and manipulating them.
  11:  *
  12:  *)
  13: 
  14: (* An interface for BDD module *)
  15:     type bdd (** Abstract type for binary decision diagrams. *)
  16:   
  17:     type expression (** Abstract type for boolean expression .*)
  18: 
  19:     type operation (** Abstract type for boolean operators *)
  20: 
  21:     (**
  22:      Make builds a node out of an expression and ensures the bdd is reduced
  23:      as outlined in 'An Introduction to Binary Decision Diagrams' 
  24:      by Andersen. It returns that node, as well. 
  25:      *)
  26:     val make: int -> bdd -> bdd -> bdd
  27: 
  28:     (**
  29:      Build takes an expression and recursively constructs its binary decision 
  30:      diagram. It primarily uses Make for constructing each Node. Given 
  31:      expression should have variables given in order 1, 2, 3... if the user
  32:      wants correct sat_count solutions.
  33:      *)
  34:     val build: expression -> bdd
  35:     
  36:     (**
  37:      Apply takes a boolean operator of type operation and two bdds, and 
  38:      returns a new bdd that is the result of the given operation.
  39:      *)
  40:     val apply: operation -> bdd -> bdd -> bdd
  41: 
  42:    (**
  43:     Sat_count returns the number of solutions to the Boolean function
  44:     given f(x) = 1. 
  45:     *)
  46:     val sat_count: bdd -> float
  47: 
  48:     (**
  49:      Any_sat returns a list of variable assignments that result in f(x) = 1. 
  50:      The path any_sat takes is greedy, and it may fail in the case of functions
  51:      that aren't well behaved. In those cases, NoSolutionFound exception is 
  52:      thrown and all_sat should be used to find the solution.
  53:      *)
  54:     val any_sat: bdd -> (int * expression) list
  55: 
  56:     (**
  57:      All_sat returns all satisfying variable assignments that result in f(x)=1 
  58:      where f(x) is the boolean expression for the given bdd. 
  59:      *)
  60:     val all_sat: bdd -> (int * expression) list list
  61: 
  62:     (** 
  63:      restrict takes a variable assignment and a bdd and returns the new, 
  64:      simplified bdd. Variable assignments should be passed as (var number, 
  65:      True/False). 
  66:      *)
  67:     val restrict: bdd -> (expression * expression) -> bdd

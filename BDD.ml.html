  14: (* An implementation of the BDD module *)
  15: (*module BDD =
  16:   struct*)
  17:     (* abstract data type for expressions *)
  18:     type expression =
  19:       | False
  20:       | True
  21:       | Var of int
  22:       | And of expression * expression (* && *)
  23:       | Or of expression * expression (* || *)
  24:       | Imp of expression * expression (* ->, implication *)
  25:       | BImp of expression * expression (* <->, bi-implication *)
  26:       | Neg of expression (* -, negation *)
  27:     
  28:     (* abstract data type for Boolean operations, used in apply *)
  29:     type operation = 
  30:       | OpAnd
  31:       | OpOr
  32:       | OpImp
  33:       | OpBimp
  34:     ;;
  35: 
  36:     (* 
  37:      * BDD is represented as a binary tree with the left branch being the 
  38:      * low branch, or the branch depicting 0 value for the variable, and
  39:      * the right branch being the high branch
  40:      *)
  41:     type bdd = Zero | One | Node of int * bdd * bdd
  42: 
  43:     (* t is a hash table that holds our bdd *)
  44:     let t = Hashtbl.create 15;;
  45: 
  46:     (* initialize t to hold Zero and One, the terminal nodes *)
  47:     (*let _ = Hashtbl.add t 0 Zero;;
  48:     let _ = Hashtbl.add t 1 One;;*)
  49: 
  50:     (* h is a lookup table that will be checked to ensure the bdd constructed
  51:      * is also reduced *)
  52:     let h = Hashtbl.create 15;;
  53:     
  54:     (* increments an index given to each entry in the hash table *)
  55:     let increment = let u = ref 1 in fun () -> incr u; !u;;
  56: 
  57:     (* exception thrown by var if a terminal node is reached *)
  58:     exception TerminalNode;;
  59: 
  60:     (* returns the variable number of a node *)
  61:     let var (u : int) : int = 
  62:       match Hashtbl.find t u with
  63:         | Node(i, _, _) -> i
  64:         | Zero | One -> raise TerminalNode
  65:     ;;
  66: 
  67:     (* returns the low branch of a node *)
  68:     let low (u : int) : bdd = 
  69:       match Hashtbl.find t u with
  70:         | Zero -> Zero
  71:         | One -> One
  72:         | Node(_, l, _) -> l
  73:     ;;
  74:             
  75:     (* returns the high branch of a node *)
  76:     let high (u : int) : bdd = 
  77:       match Hashtbl.find t u with
  78:         | Zero -> Zero
  79:         | One -> One
  80:         | Node(_, _, h) -> h
  81:     ;;
  82: 
  83:     (* eval reduces the boolean expression at each step of building the bdd. 
  84:      * If the current expression is still not completely filled with True or 
  85:      *  False, eval returns the unaltered expression. *)
  86:     let rec eval (exp: expression) : expression = 
  87:       match exp with
  88:         | True -> True
  89:         | False -> False
  90:         | Var i -> Var i
  91:         | Neg x -> (
  92:             match x with
  93:               | True -> False
  94:               | False -> True
  95:               | _ -> Neg (eval x)
  96:           )
  97:         | And(x,y) -> (
  98:             match (x,y) with
  99:               | (True,True) -> True
 100:               | (True,False) -> False
 101:               | (False, True) -> False
 102:               | (False, False) -> False
 103:               | (_, _) -> And(eval x, eval y)
 104:           )
 105:         | Or(x,y) ->  (
 106:             match (x,y) with 
 107:               | (True,True) -> True
 108:               | (True, False) -> True
 109:               | (False, True) -> True
 110:               | (False, False) -> False
 111:               | (_, _) -> Or(eval x,eval y)
 112:           )
 113:         | Imp(x,y) -> (
 114:             match (x,y) with 
 115:               | (True,True) -> True 
 116:               | (True,False) -> False
 117:               | (False, True) -> True
 118:               | (False, False) -> True
 119:               | (_, _) -> Imp(eval x,eval y)
 120:           )
 121:         | BImp(x,y) -> (
 122:             match (x,y) with
 123:               | (True,True) -> True
 124:               | (True,False) -> False 
 125:               | (False, True) -> False
 126:               | (False, False) -> True
 127:               | (_, _) -> BImp(eval x,eval y)
 128:           )
 129:     ;;
 130:     
 131:     (* var_bool checks whether a variable is present in the current expression
 132:      * Revisit: Is this necessary if variables are in correct (1, 2, 3...) 
 133:      * ordering. I currently don't think so. 
 134:      *)
 135:     let rec var_bool (v : expression) (e : expression) : bool = 
 136:       match e with
 137:         | True -> false
 138:         | False -> false
 139:         | Var i -> let Var j = v in j == i (* only important part *)
 140:         | And(x, y) -> var_bool v x || var_bool v y
 141:         | Or(x, y) -> var_bool v x || var_bool v y
 142:         | BImp(x, y) -> var_bool v x || var_bool v y
 143:         | Imp(x, y) -> var_bool v x || var_bool v y
 144:         | Neg x -> var_bool v x;;
 145: 
 146:     (* returns an expression with the variable v replaced by asgmt 
 147:      * This function became necessary after we realized that Boolean 
 148:      * expressions may have repeating variables *)
 149:     let rec var_lookup (v : expression) (e : expression) (asgmt : expression) 
 150:         : expression = 
 151:       match e with
 152:         | True -> True
 153:         | False -> False
 154:         | Var i -> let Var j = v in if j == i then asgmt else e
 155:         | And(x, y) -> And (var_lookup v x asgmt, var_lookup v y asgmt)
 156:         | Or(x, y) -> Or (var_lookup v x asgmt, var_lookup v y asgmt)
 157:         | BImp(x, y) -> BImp (var_lookup v x asgmt, var_lookup v y asgmt)
 158:         | Imp(x, y) -> Imp (var_lookup v x asgmt, var_lookup v y asgmt)
 159:         | Neg x -> Neg (var_lookup v x asgmt);;
 160: 
 161:     (* make creates each node
 162:      * First, it checks whether hash table h already holds a node with the
 163:      * attributes i, low, and high. If yes, then nothing is added to t, and 
 164:      * the node is returned. If not, the node is added to h and t. t holds
 165:      * the actual bdd while h is just a lookup table. 
 166:      *)
 167:     let make (i :int) (low : bdd) (high : bdd) : bdd=
 168:       if low == high then low
 169:       else if Hashtbl.mem h (Node(i, low, high))
 170:       then Node(i, low, high)
 171:       else 
 172:         let u = increment () in 
 173:         let _ = Hashtbl.add h (Node(i, low, high)) u in
 174:         let _ = Hashtbl.add t u (Node(i, low, high)) in
 175:             Node(i, low, high)
 176:     ;;
 177:     
 178:     (*
 179:      * build actually creates the bdd after it gets an expression
 180:      * It does so recursively by calling itself and passing shannon either
 181:      * True or False. 
 182:      * Example: build (And (BImp (Var 1, Var 2), BImp (Var 3, Var 4)))
 183:      *            =  Node (1,Node (2, Node (3, Node (4, One, Zero), 
 184:      *                 Node (4, Zero, One)), Zero),
 185:      *                   Node (2, Zero, Node (3, Node (4, One, Zero), 
 186:      *                     Node (4, Zero, One))))
 187:      *) 
 188:     let build (exp : expression) : bdd = 
 189:       let rec build' (e : expression) (i : int) : bdd =
 190:         match e with 
 191:           | True -> 
 192:               (* Add One to the hash table *)
 193:               if Hashtbl.mem h (One) then One
 194:               else
 195:                 let _ = Hashtbl.add h (One) (1) in 
 196:                 let _ = Hashtbl.add t 1 (One) in One
 197:           | False -> 
 198:               (* Add Zero to the hash table *)
 199:               if Hashtbl.mem h (Zero) then Zero
 200:               else
 201:                 let _ = Hashtbl.add h (Zero) (0) in 
 202:                 let _ = Hashtbl.add t 0 (Zero) in Zero
 203:           | _ -> 
 204:               (* Check if the variable i is in the expression, move on if not*)
 205:               if var_bool (Var i) e then
 206:                 (* recursively call build on low and high branches after var i
 207:                  * is replaced with True for high branch and False for low
 208:                  * branch *)
 209:                 let low = 
 210:                   build' (eval (var_lookup (Var i) e False)) (i+1) in
 211:                 let high = build' (eval (var_lookup (Var i) e True)) (i+1) in
 212:                   make (i) (low) (high)
 213:               else build' (eval e) (i + 1) (* make creates the node *)
 214:       in build' exp 1
 215:     ;;

 217:     (* 
 218:      * sat_count returns a count of satisfying paths leading to a One
 219:      * terminal. 
 220:      *)
 221:     let sat_count (b: bdd) : float = 
 222:       try let u = Hashtbl.find h b in
 223:       let rec count (u : int) : float = 
 224:         if u == 0 then 0.
 225:         else if u == 1 then 1.
 226:         else 
 227:           let low_u = Hashtbl.find h (low u) in
 228:           let high_u = Hashtbl.find h (high u) in
 229:           try let low_var = var low_u in
 230:           let high_var = var high_u in 
 231:             ((2. ** float_of_int(low_var - (var u) - 1)) *. (count(low_u))) +. 
 232:               ((2. ** float_of_int(high_var - (var u) - 1)) *. (count(high_u)))
 233:           with TerminalNode -> 
 234:             ((2. ** float_of_int(((var u)+1)- (var u) - 1)) 
 235:              *. (count(low_u))) +. ((2. ** 
 236:             float_of_int(((var u) + 1) - (var u) - 1)) *. (count(high_u)))
 237:       in (count u) *. (2.**(float_of_int(var u) -. 1.))
 238:       with TerminalNode -> let u = Hashtbl.find h b in 
 239:         if u == 0 then 0. else 1.
 240:     ;;

 242:     (* Exception thrown in case any_sat encounters 0 *)
 243:     exception NoSolutionFound;;

 245:     (* returns a greedy solution to the BDD 
 246:      * Note: Because this algorithm is greedy, it fails to find a solution for
 247:      * functions that aren't well-behaved. Greediness gives O(n) runtime *)
 248:     let any_sat (b : bdd) : (int * expression) list  = 
 249:       let u = Hashtbl.find h b in
 250:       let rec loop (u : int) : (int * expression) list = 
 251:         let low_u = Hashtbl.find h (low u) in
 252:         let high_u = Hashtbl.find h (high u) in
 253:         if u == 0 then raise NoSolutionFound
 254:         else if u == 1 then []
 255:         else
 256:           let var = var u in 
 257:           if low_u == 0 then (var, True)::(loop (high_u))
 258:           else (var, False)::(loop(low_u))
 259:       in loop u
 260:     ;;
 261: 
 262:     (* Coolest function all_sat. Finds all satisfying assignments for the bdd *)
 263:     let all_sat (b : bdd) : (int * expression) list list =
 264:       let u = Hashtbl.find h b in
 265:       let rec all (u: int) : (int * expression) list list = 
 266:         let low_u = Hashtbl.find h (low u) in
 267:         let high_u = Hashtbl.find h (high u) in
 268:         if u == 0 then []
 269:         else if u == 1 then [[]]
 270:         else  
 271:           let var = var u in
 272:           (List.map (fun a-> (var, False)::a) (all(low_u)))@
 273:           (List.map (fun a-> (var, True)::a) (all(high_u)))
 274:       in all u
 275:     ;;

 280:     (* Apply takes two bdds and a boolean op of type operation and returns
 281:      * a new bdd that is a product of that operation. 
 282:      * Revisit: Can apply be used to construct bdds? *)
 283:     let apply (op : operation) (b1 : bdd) (b2 : bdd) : bdd =
 284:       let rec app (b1 : bdd) (b2 : bdd) : bdd =
 285:         let u1 = Hashtbl.find h b1 in
 286:         let u2 = Hashtbl.find h b2 in 
 287:         if (u1 == 0 || u1 == 1) && (u2 == 0 || u2 == 1)
 288:         then let (e1, e2) = 
 289:           match (u1, u2) with
 290:             | (0, 0) -> (False, False)
 291:             | (1, 0) -> (True, False)
 292:             | (0, 1) -> (False, True)
 293:             | (1, 1) -> (True, True) 
 294:         in match op with 
 295:           | OpAnd -> 
 296:               (match eval (And (e1, e2)) with 
 297:                  | True -> One
 298:                  | False -> Zero
 299:                  | _ -> raise InvalidInput)
 300:           | OpOr ->
 301:               (match eval (Or (e1, e2)) with 
 302:                  | True -> One
 303:                  | False -> Zero
 304:                  | _ -> raise InvalidInput)               
 305:           | OpBimp ->
 306:               (match eval (BImp (e1, e2)) with 
 307:                  | True -> One
 308:                  | False -> Zero
 309:                  | _ -> raise InvalidInput)
 310:           | OpImp ->
 311:               (match eval (Imp (e1, e2)) with 
 312:                  | True -> One
 313:                  | False -> Zero
 314:                  | _ -> raise InvalidInput) 
 315:         else if (u1 == 0) || (u1 == 1)
 316:         then make (var u2) (app b1 (low u2)) (app b1 (high u2))
 317:         else if (u2 == 0) || (u2 == 1)
 318:         then make (var u1) (app (low u1) b2) (app (high u1) b2)
 319:         else
 320:           if (var u1) == (var u2) 
 321:           then make (var u1) (app (low u1) (low u2)) (app (high u1) (high u2))
 322:           else if var u1 < var u2
 323:           then make (var u1) (app (low u1) b2) (app (high u1) b2)
 324:           else (* var u1 > var u2 *)
 325:             make (var u2) (app b1 (low u2)) (app b1 (high u2))
 326:       in app b1 b2;;

 328:     (* restrict takes a variable and a variable assignment and returns a 
 329:      * restructured bdd with the given variable factored out. Also, really 
 330:      * cool. *)
 331:     let restrict (b: bdd) (asn: expression * expression) : bdd =
 332:       let init = Hashtbl.find h b in
 333:       let rec res (u: int) : bdd = (
 334:       match asn with
 335:         | (Var j, z) -> 
 336:               (* detects base cases *)
 337:               if u == 0 then Zero
 338:               else if(u == 1) then One
 339:               (* traverses BDD *)
 340:               else  
 341:                 try 
 342:                   let j1 = var j in
 343:                   let v1 = var u in
 344:                   let low_u = Hashtbl.find h (low u) in
 345:                   let high_u = Hashtbl.find h (high u) in 
 346:                   if(v1 > j1) then Hashtbl.find t u
 347:                   else if(v1 < j1) then make (v1) (res (low_u)) (res (high_u))         
 348:                   else 
 349:                     if(z == False) then res (low_u)
 350:                     else res (high_u)
 351:                 (* if we encounter a terminal node *)
 352:                 with TerminalNode -> 
 353:                   if(z == False) then low u
 354:                   else high u
 355:         | _ -> raise InvalidInput )
 356:         in
 357:         res (init)
 358:     ;;
